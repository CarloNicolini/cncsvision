set(QTMOTORS_HDR InfoDialog.h  MainWindow.h  SpeedBox.h)
set(QTMOTORS_SRC InfoDialog.cpp  main.cpp  MainWindow.cpp  SpeedBox.cpp )
set(QTMOTORS_FORMS InfoDialog.ui  MainWindow.ui )
set(QTMOTORS_RESOURCES "resources.qrc")
link_libraries(${QT_QTCORE_LIBRARY} ${QT_QTGUI_LIBRARY} ${Boost_LIBRARIES})
include_directories(${QT_INCLUDE_PATH} ${QT_QTGUI_INCLUDE_DIR} ${QT_QTCORE_INCLUDE_DIR} ${CMAKE_SOURCE_DIR}/src/core ${CMAKE_SOURCE_DIR}/src/experiment ${CMAKE_SOURCE_DIR}/src/eigen )

include_directories(${CMAKE_CURRENT_SOURCE_DIR} ${CMAKE_CURRENT_BINARY_DIR} ${QT_INCLUDES})

QT4_ADD_RESOURCES( QTMOTORS_RESOURCES_SOURCES ${QTMOTORS_RESOURCES} )
QT4_WRAP_UI( QTMOTORS_FORMS_HEADERS ${QTMOTORS_FORMS} )
QT4_WRAP_CPP( QTMOTORS_HEADERS_MOC ${QTMOTORS_HDR} )

add_executable(QtMotors WIN32 MACOSX_BUNDLE  ${QTMOTORS_SRC} ${QTMOTORS_HDR} ${QTMOTORS_HEADERS_MOC}
	${QTMOTORS_RESOURCES_SOURCES} ${QTMOTORS_FORMS_HEADERS} ${QTMOTORS_RCS} )
target_link_libraries(QtMotors ${QT_LIBRARIES} Communication ${Boost_LIBRARIES})
